In `components/NavDropdowns.tsx`, fix the outside-click event listener so it fires on both mouse and touch input.

Change the `useEffect` in `NavDropdowns`:

FROM:
```tsx
function handleOutsideClick(e: MouseEvent) {
  if (navRef.current && !navRef.current.contains(e.target as Node)) {
    setOpen(null);
  }
}
document.addEventListener('mousedown', handleOutsideClick);
return () => document.removeEventListener('mousedown', handleOutsideClick);
```

TO:
```tsx
function handleOutsideClick(e: PointerEvent) {
  if (navRef.current && !navRef.current.contains(e.target as Node)) {
    setOpen(null);
  }
}
document.addEventListener('pointerdown', handleOutsideClick);
return () => document.removeEventListener('pointerdown', handleOutsideClick);
```

This is the complete scope of the change to NavDropdowns.tsx. The `onMouseEnter`/`onMouseLeave` handlers on the desktop nav groups remain — they are desktop-only (the mobile nav is in Header.tsx) and acceptable for md+ viewports where mouse input is expected.

No other changes to NavDropdowns.tsx.
